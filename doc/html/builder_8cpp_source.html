50  DisjointSet<Cluster *> U;
51  std::vector<std::pair<double, std::pair<Cluster *, Cluster *> > > E;
52  BOOST_FOREACH (ClusterPtr cls, data_) {
53  U.Add(cls.get());
54  BOOST_FOREACH (Cluster *adj, cls->adjacent_) {
55  double strength = (cls->radius_ + adj->radius_) / 2.0;
56  Cluster *a = cls.get(), *b = adj;
57  if (a < b)
58  E.push_back(std::make_pair(strength, std::make_pair(cls.get(), adj)));
59  }
60  }

65  Cluster *a = it.second.first;
66  Cluster *b = it.second.second;
67  if (U.IsSame(a, b)) {

113 void Builder::ComputeRadius() {
